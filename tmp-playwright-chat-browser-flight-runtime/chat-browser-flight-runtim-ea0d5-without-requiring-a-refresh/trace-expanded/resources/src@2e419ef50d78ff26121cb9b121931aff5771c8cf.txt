import {
  closeElectronApp,
  expect,
  getStableWindow,
  installIpcMocks,
  test,
} from './fixtures/electron';

const SESSION_KEY = 'agent:main:main';
const SESSION_ID = 'browser-flight-runtime';
const RUN_ID = 'run-browser-flight-runtime';
const PROMPT = '打开浏览器，访问携程，查一下明天深圳到南京的机票，列一个表格给我';
const FINAL_TABLE_HEADER = '| 航空公司 | 航班 | 起飞 | 到达 | 价格 |';
const FINAL_ROW_TEXT = '| 东方航空 | MU2878 | 23:25 | 01:50(+1天) | ￥620 起 |';

function stableStringify(value: unknown): string {
  if (value == null || typeof value !== 'object') return JSON.stringify(value);
  if (Array.isArray(value)) return `[${value.map((item) => stableStringify(item)).join(',')}]`;
  const entries = Object.entries(value as Record<string, unknown>)
    .sort(([left], [right]) => left.localeCompare(right))
    .map(([key, entryValue]) => `${JSON.stringify(key)}:${stableStringify(entryValue)}`);
  return `{${entries.join(',')}}`;
}

async function emitGatewayNotification(
  app: Parameters<typeof installIpcMocks>[0],
  payload: unknown,
): Promise<void> {
  await app.evaluate(({ BrowserWindow }, nextPayload) => {
    const window = BrowserWindow.getAllWindows().at(-1);
    if (!window) throw new Error('No BrowserWindow available');
    window.webContents.send('gateway:notification', nextPayload);
  }, payload);
}

test.describe('Chat browser flight runtime', () => {
  test('finishes the running state after a browser-style flight table reply without requiring a refresh', async ({ launchElectronApp }) => {
    test.setTimeout(180_000);

    const app = await launchElectronApp({ skipSetup: true });

    try {
      await installIpcMocks(app, {
        gatewayStatus: { state: 'running', port: 18789, pid: 12345, connectedAt: Date.now() },
        hostApi: {
          [stableStringify(['/api/gateway/status', 'GET'])]: {
            ok: true,
            data: {
              status: 200,
              ok: true,
              json: {
                state: 'running',
                port: 18789,
                pid: 12345,
                connectedAt: Date.now(),
              },
            },
          },
          [stableStringify(['/api/agents', 'GET'])]: {
            ok: true,
            data: {
              status: 200,
              ok: true,
              json: {
                success: true,
                agents: [],
              },
            },
          },
        },
      });

      await app.evaluate(({ ipcMain, BrowserWindow }, { prompt, sessionKey, sessionId, runId, finalHeader, finalRow }) => {
        let sessions: Array<{ key: string; id: string; label: string; updatedAt: number }> = [];
        let historyMessages: Array<Record<string, unknown>> = [];

        function emitNotification(payload: unknown): void {
          const window = BrowserWindow.getAllWindows().at(-1);
          if (!window) throw new Error('No BrowserWindow available');
          window.webContents.send('gateway:notification', payload);
        }

        ipcMain.removeHandler('gateway:rpc');
        ipcMain.handle('gateway:rpc', async (_event, method: string, params?: { sessionKey?: string; message?: string }) => {
          if (method === 'sessions.list') {
            return { sessions };
          }

          if (method === 'chat.history') {
            return { messages: historyMessages };
          }

          if (method === 'chat.abort') {
            return { ok: true };
          }

          if (method === 'chat.send') {
            const now = Date.now();
            const activeSessionKey = params?.sessionKey || sessionKey;
            const visiblePrompt = prompt;

            sessions = [{
              key: activeSessionKey,
              id: sessionId,
              label: visiblePrompt,
              updatedAt: now,
            }];

            historyMessages = [{
              id: 'user-browser-flight-1',
              role: 'user',
              content: visiblePrompt,
              timestamp: Math.floor(now / 1000),
            }];

            setTimeout(() => {
              emitNotification({
                method: 'agent',
                params: {
                  phase: 'started',
                  runId,
                  sessionKey: activeSessionKey,
                },
              });
            }, 0);

            setTimeout(() => {
              emitNotification({
                method: 'agent',
                params: {
                  runId,
                  sessionKey: activeSessionKey,
                  stream: 'item',
                  data: {
                    itemId: 'tool:browser-flight-1',
                    phase: 'start',
                    kind: 'tool',
                    name: 'browser',
                    status: 'running',
                    title: '打开浏览器',
                    progressText: '正在访问携程并查询深圳到南京的机票',
                    toolCallId: 'browser-flight-1',
                  },
                },
              });
            }, 300);

            setTimeout(() => {
              emitNotification({
                method: 'agent',
                params: {
                  runId,
                  sessionKey: activeSessionKey,
                  stream: 'assistant',
                  data: {
                    text: '我先打开浏览器访问携程，再整理明天深圳到南京的机票信息。',
                    delta: '我先打开浏览器访问携程，再整理明天深圳到南京的机票信息。',
                  },
                },
              });
            }, 700);

            setTimeout(() => {
              const completedAt = Date.now();
              historyMessages = [
                {
                  id: 'user-browser-flight-1',
                  role: 'user',
                  content: visiblePrompt,
                  timestamp: Math.floor(now / 1000),
                },
                {
                  id: 'assistant-browser-flight-1',
                  role: 'assistant',
                  content: [
                    '已帮你整理明天深圳到南京的机票信息：',
                    '',
                    finalHeader,
                    '| --- | --- | --- | --- | --- |',
                    finalRow,
                    '| 湖南航空 | A67297 | 07:15 | 09:30 | ￥650 起 |',
                    '| 深圳航空 | ZH9847 | 14:35 | 17:00 | ￥750 起 |',
                    '',
                    '如果需要，我可以继续帮你按最早起飞、最低价格或白天时段再筛一遍。',
                  ].join('\n'),
                  timestamp: Math.floor(completedAt / 1000),
                },
              ];
              sessions = [{
                key: activeSessionKey,
                id: sessionId,
                label: visiblePrompt,
                updatedAt: completedAt,
              }];

              emitNotification({
                method: 'agent',
                params: {
                  phase: 'completed',
                  runId,
                  sessionKey: activeSessionKey,
                },
              });
            }, 2_800);

            return {
              success: true,
              result: {
                runId,
              },
            };
          }

          return {};
        });
      }, {
        prompt: PROMPT,
        sessionKey: SESSION_KEY,
        sessionId: SESSION_ID,
        runId: RUN_ID,
        finalHeader: FINAL_TABLE_HEADER,
        finalRow: FINAL_ROW_TEXT,
      });

      const page = await getStableWindow(app);
      await expect(page.getByTestId('main-layout')).toBeVisible({ timeout: 60_000 });

      const composer = page.getByTestId('chat-composer');
      const messageInput = composer.getByRole('textbox');
      const sendButton = composer.getByTestId('chat-send-button');

      await messageInput.fill(PROMPT);
      await sendButton.click();

      const sessionRow = page.getByTestId(`sidebar-session-${SESSION_KEY}`);
      await expect(sessionRow).toBeVisible({ timeout: 30_000 });
      await expect(page.getByTestId(`sidebar-session-running-indicator-${SESSION_KEY}`)).toBeVisible({ timeout: 30_000 });

      await expect.poll(async () => {
        return await sendButton.evaluate((node) => !!node.querySelector('svg.lucide-square'));
      }, { timeout: 30_000 }).toBe(true);

      await expect(page.getByText('正在访问携程并查询深圳到南京的机票')).toBeVisible({ timeout: 30_000 });

      await expect(page.getByText(FINAL_TABLE_HEADER, { exact: true })).toBeVisible({ timeout: 60_000 });
      await expect(page.getByText(FINAL_ROW_TEXT, { exact: true })).toBeVisible({ timeout: 60_000 });

      await expect(page.getByTestId(`sidebar-session-running-indicator-${SESSION_KEY}`)).toHaveCount(0, { timeout: 20_000 });
      await expect.poll(async () => {
        return await sendButton.evaluate((node) => !!node.querySelector('svg.lucide-send-horizontal'));
      }, { timeout: 20_000 }).toBe(true);

      const sendButtonTitle = await sendButton.getAttribute('title');
      expect(['Send', '发送']).toContain(sendButtonTitle);

      await page.waitForTimeout(1_500);
      await expect(page.getByTestId(`sidebar-session-running-indicator-${SESSION_KEY}`)).toHaveCount(0);
      await expect.poll(async () => {
        return await sendButton.evaluate((node) => !!node.querySelector('svg.lucide-send-horizontal'));
      }).toBe(true);
    } finally {
      await closeElectronApp(app);
    }
  });
});
